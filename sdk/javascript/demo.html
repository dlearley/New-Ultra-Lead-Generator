Basic Setup

<script src="ultralead.js"></script>
<script>
  UltraLead.init({
    apiKey: 'YOUR_API_KEY',
    organizationId: 'YOUR_ORG_ID',
  });
</script>

Custom Form Submission

UltraLead.submitForm('form-id', {
  email: 'john@example.com',
  firstName: 'John',
  lastName: 'Doe',
});

Visitor Identification

UltraLead.identify('john@example.com', {
  firstName: 'John',
  company: 'Acme Inc',
});
